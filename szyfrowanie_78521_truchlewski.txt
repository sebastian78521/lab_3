import base64

wiadomosc = "kodowanie testowe"
wiadomosc_bytes = wiadomosc.encode('ascii')
base64_bytes = base64.b64encode(wiadomosc_bytes)
base64_wiadomosc = base64_bytes.decode('ascii')

print(base64_wiadomosc)
